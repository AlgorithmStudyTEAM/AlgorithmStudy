class Solution {

    // 1 1 1 1 1 1
    // 1 2 3 4 5 6 
    // 1 3 6 10 15 21 
    // 1 4 10 20 35 56 

    public int uniquePaths(int m, int n) {
        int[][] p = new int[m][n];

        for(int y=0; y<m; y++){
            for(int x=0; x<n; x++){
                if(x==0||y==0){
                    p[y][x] = 1;
                }else{
                    p[y][x] = p[y-1][x] + p[y][x-1];
                }
            }
        }

        return p[m-1][n-1];
    }
